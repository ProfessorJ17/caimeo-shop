  <script type="module">
  // Import noble-post-quantum modules from jsDelivr (client-side ES module).
  // Uses paulmillr's noble-post-quantum library (pure JS, supports deterministic keygen via seed).
  import { ml_kem512, ml_kem768, ml_kem1024 } from 'https://esm.sh/@noble/post-quantum@0.5.1/ml-kem.js?bundle';
  import { ml_dsa44, ml_dsa65, ml_dsa87 } from 'https://esm.sh/@noble/post-quantum@0.5.1/ml-dsa.js?bundle';

  // Utilities
  const out = document.getElementById('output');
  function log(...t){ out.textContent = (new Date()).toLocaleTimeString() + ' | ' + t.join(' ') + '\n' + out.textContent; }
  function bytesToB64(bytes){
    // safe chunked conversion
    let CHUNK = 0x8000;
    let str = '';
    for (let i=0; i<bytes.length; i += CHUNK) {
      str += String.fromCharCode.apply(null, bytes.subarray(i, i+CHUNK));
    }
    return btoa(str);
  }
  function equalBytes(a,b){
    if(a.length !== b.length) return false;
    for(let i=0;i<a.length;i++) if(a[i] !== b[i]) return false;
    return true;
  }
  async function deriveSeed(pass, salt, iterations=120000, outBytes=64){
    const enc = new TextEncoder();
    const key = await crypto.subtle.importKey('raw', enc.encode(pass), {name:'PBKDF2'}, false, ['deriveBits']);
    const bits = await crypto.subtle.deriveBits({
      name:'PBKDF2',
      salt: enc.encode(salt || ''),
      iterations: Number(iterations),
      hash: 'SHA-256'
    }, key, outBytes * 8);
    return new Uint8Array(bits);
  }

  // Mapping selectors
  const kemMap = { ml_kem512, ml_kem768, ml_kem1024 };
  const dsaMap = { ml_dsa44, ml_dsa65, ml_dsa87 };

  // Generate and test KEM (encap/decap)
  async function genAndTestKEM(algKey, pass, salt, iterations){
    try{
      log('Deriving seed...');
      const seed = await deriveSeed(pass, salt, iterations, 64); // 64 bytes recommended
      log('Seed derived (hex prefix):', Array.from(seed.slice(0,8)).map(x=>x.toString(16).padStart(2,'0')).join(''));
      const kem = kemMap[algKey];
      if(!kem) throw new Error('KEM not found: ' + algKey);

      log('Key generation (deterministic) using', algKey);
      const keys = kem.keygen(seed); // returns {publicKey, secretKey} (Uint8Array)
      const pubB64 = bytesToB64(keys.publicKey);
      const privB64 = bytesToB64(keys.secretKey);
      log('Public key (base64, prefix):', pubB64.slice(0,48) + '...');
      log('Private key (base64, prefix):', privB64.slice(0,48) + '...');

      // Quick self-test: encapsulate -> decapsulate and compare shared secrets
      log('Encapsulating to public key...');
      const { cipherText, sharedSecret: bobShared } = kem.encapsulate(keys.publicKey);
      log('Decapsulating with secret key...');
      const aliceShared = kem.decapsulate(cipherText, keys.secretKey);

      const ok = equalBytes(aliceShared, bobShared);
      log('KEM test result:', ok ? 'OK — shared secrets match' : 'FAIL — mismatch');

      return { keys, pubB64, privB64, ok, aliceShared, bobShared };
    }catch(e){
      log('ERROR in KEM:', e.message || e);
      throw e;
    }
  }

  // Generate and test DSA (sign/verify)
  async function genAndTestDSA(algKey, pass, salt, iterations){
    try{
      log('Deriving seed for DSA...');
      const seed = await deriveSeed(pass, salt, iterations, 32); // ML-DSA expects 32-byte seed
      const dsa = dsaMap[algKey];
      if(!dsa) throw new Error('DSA not found: ' + algKey);

      log('DSA keygen (deterministic) using', algKey);
      const keys = dsa.keygen(seed); // returns {publicKey, secretKey}
      const pubB64 = bytesToB64(keys.publicKey);
      const privB64 = bytesToB64(keys.secretKey);
      log('Public key (b64 prefix):', pubB64.slice(0,48) + '...');
      log('Private key (b64 prefix):', privB64.slice(0,48) + '...');

      // Test sign/verify
      const msg = new TextEncoder().encode('noble-pq demo test message');
      // API: ml_dsa.sign(secretKey, msg) -> signature (Uint8Array)
      // and ml_dsa.verify(publicKey, msg, signature) -> boolean
      // (noble-post-quantum exposes sign(secretKey, message) in examples and literature)
      log('Signing test message...');
      const sig = dsa.sign(msg, keys.secretKey);
      log('Verifying signature...');
      const verified = dsa.verify(sig, msg, keys.publicKey);
      log('DSA test result:', verified ? 'OK — signature verified' : 'FAIL — verification failed');
      return { keys, pubB64, privB64, verified };
    }catch(e){
      log('ERROR in DSA:', e.message || e);
      throw e;
    }
  }

  // Hook UI
  document.getElementById('genKem').onclick = async () => {
    const pass = document.getElementById('pass').value;
    if(!pass){ alert('Enter passphrase'); return; }
    const salt = document.getElementById('salt').value;
    const iters = Number(document.getElementById('iters').value || 120000);
    const algKey = document.getElementById('kemSelect').value;
    try{
      window._lastKemResult = await genAndTestKEM(algKey, pass, salt, iters);
      const r = window._lastKemResult;
      if (r) { 
        localStorage.setItem('pq.kem.pub', r.pubB64||''); 
        localStorage.setItem('pq.kem.priv', r.privB64||''); 
        localStorage.setItem('pq.hasKeys','1'); 
        // Notify opener (main app) if present so it can save public key to cloud immediately
        try { 
          if (window.opener && !window.opener.closed) {
            window.opener.postMessage({ type: 'pq:keysGenerated', kemPub: r.pubB64||'', dsaPub: null }, '*');
          }
        } catch(_) {}
      }
    }catch(e){}
  };

  document.getElementById('genDsa').onclick = async () => {
    const pass = document.getElementById('pass').value;
    if(!pass){ alert('Enter passphrase'); return; }
    const salt = document.getElementById('salt').value;
    const iters = Number(document.getElementById('iters').value || 120000);
    const algKey = document.getElementById('dsaSelect').value;
    try{
      window._lastDsaResult = await genAndTestDSA(algKey, pass, salt, iters);
      const r = window._lastDsaResult;
      if (r) { 
        localStorage.setItem('pq.dsa.pub', r.pubB64||''); 
        localStorage.setItem('pq.dsa.priv', r.privB64||''); 
        localStorage.setItem('pq.hasKeys','1'); 
        // Notify opener (main app) if present so it can save public key to cloud immediately
        try { 
          if (window.opener && !window.opener.closed) {
            window.opener.postMessage({ type: 'pq:keysGenerated', kemPub: null, dsaPub: r.pubB64||'' }, '*');
          }
        } catch(_) {}
      }
    }catch(e){}
  };

  // Download helpers
  function downloadText(filename, text){
    const blob = new Blob([text], {type:'application/octet-stream'});
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = filename;
    document.body.appendChild(a);
    a.click();
    a.remove();
    URL.revokeObjectURL(url);
  }
  document.getElementById('downloadKemPub').onclick = () => {
    const r = window._lastKemResult;
    if(!r) { alert('Generate KEM keys first'); return; }
    downloadText('kem_public.b64', r.pubB64);
  };
  document.getElementById('downloadKemPriv').onclick = () => {
    const r = window._lastKemResult;
    if(!r) { alert('Generate KEM keys first'); return; }
    if(!confirm('Download private key? Keep it secret.')) return;
    downloadText('kem_private.b64', r.privB64);
  };
  document.getElementById('downloadDsaPub').onclick = () => {
    const r = window._lastDsaResult;
    if(!r) { alert('Generate DSA keys first'); return; }
    downloadText('dsa_public.b64', r.pubB64);
  };
  document.getElementById('downloadDsaPriv').onclick = () => {
    const r = window._lastDsaResult;
    if(!r) { alert('Generate DSA keys first'); return; }
    if(!confirm('Download private key? Keep it secret.')) return;
    downloadText('dsa_private.b64', r.privB64);
  };

  // Auto-test loop (KEM)
  let autoStop = false;
  document.getElementById('startAuto').onclick = async () => {
    const pass = document.getElementById('pass').value;
    if(!pass){ alert('Enter passphrase'); return; }
    const salt = document.getElementById('salt').value;
    const iters = Number(document.getElementById('iters').value || 120000);
    const algKey = document.getElementById('kemSelect').value;
    const total = Number(document.getElementById('autoCount').value) || 100;
    const delay = Number(document.getElementById('autoDelay').value) || 300;
    autoStop = false;
    document.getElementById('startAuto').disabled = true;
    document.getElementById('stopAuto').disabled = false;
    log('AUTO-TEST START (KEM):', algKey, 'iterations=', total, 'delay(ms)=', delay);
    let passCount = 0, failCount = 0;
    for(let i=1;i<=total;i++){
      if(autoStop) break;
      log(`Auto test #${i} — starting`);
      try{
        const res = await genAndTestKEM(algKey, pass, salt, iters);
        if(res.ok) passCount++; else failCount++;
      }catch(err){
        failCount++;
      }
      log(`Auto stats so far: passed=${passCount} failed=${failCount}`);
      await new Promise(r => setTimeout(r, delay));
    }
    log('AUTO-TEST DONE — passed=', passCount, 'failed=', failCount);
    document.getElementById('startAuto').disabled = false;
    document.getElementById('stopAuto').disabled = true;
  };
  document.getElementById('stopAuto').onclick = () => {
    autoStop = true;
    document.getElementById('stopAuto').disabled = true;
    document.getElementById('startAuto').disabled = false;
    log('AUTO-TEST STOP requested');
  };

  // Initial message
  log('Page loaded. Use passphrase+salt and press Generate.');
