    const isIPv4 = (s) => /^(\d{1,3}\.){3}\d{1,3}$/.test(s);
    const isIPv6 = (s) => /:/.test(s) && /^[0-9a-fA-F:]+$/.test(s);
    function extractHostname(input) {
      try {
        if (isIPv4(input) || isIPv6(input)) return input;
        const guess = input.includes("://") ? input : `http://${input}`;
        const u = new URL(guess);
        return u.hostname.replace(/\.$/, "");
      } catch { return input.trim(); }
    }

    async function dohResolve(hostname, type, depth = 0) {
      const endpoints = [
        `https://cloudflare-dns.com/dns-query?name=${encodeURIComponent(hostname)}&type=${type}`,
        `https://dns.google/resolve?name=${encodeURIComponent(hostname)}&type=${type}`
      ];
      const headers = { Accept: "application/dns-json" };
      const fetches = endpoints.map((u) =>
        fetch(u, { headers }).then(r => r.ok ? r.json() : Promise.reject()).catch(() => null)
      );
      const results = await Promise.all(fetches);
      const addrs = [];
      const cnames = [];
      for (const res of results) {
        if (!res) continue;
        const rr = res.Answer || res.answers || [];
        for (const a of rr) {
          if (typeof a.data === "string") {
            if (a.type === 5) cnames.push(a.data.replace(/\.$/, ""));
            if (type === "A" && isIPv4(a.data)) addrs.push(a.data);
            if (type === "AAAA" && isIPv6(a.data)) addrs.push(a.data);
          } else if (a.data && typeof a.data === "object" && a.data.ip) {
            addrs.push(a.data.ip);
          }
        }
      }
      if (addrs.length === 0 && cnames.length && depth < 3) return await dohResolve(cnames[0], type, depth + 1);
      return uniq(addrs);
    }

    function bytesHex(bytes){ return [...bytes].map(b=>b.toString(16).padStart(2,"0")).join(""); }
    async function leetIPv6FromText(text){
      const data=new TextEncoder().encode(text); const h=new Uint8Array(await crypto.subtle.digest("SHA-256",data));
      const hex=bytesHex(h); const parts=[]; for(let i=0;i<6;i++){ const start=(i*3)%hex.length; parts.push(hex.slice(start,start+4)); }
      const w1=L33T_HEX_WORDS[h[0]%L33T_HEX_WORDS.length], w2=L33T_HEX_WORDS[h[5]%L33T_HEX_WORDS.length];
      const left=[parts[0],parts[1],w1], right=[w2,parts[4],parts[5]];
      return `${left.join(":")}::${right.join(":")}`;
    }
    function leetIPv6Random(){ const r=new Uint8Array(32); crypto.getRandomValues(r); return leetIPv6FromText(bytesHex(r)); }

    async function leetFromText(text){
      const enc=new TextEncoder(); const data=enc.encode(text); const hash=await crypto.subtle.digest("SHA-256",data); return leetFromBytes(new Uint8Array(hash));
    }
    function leetFromBytes(bytes){
      const hex=[...bytes].map(b=>b.toString(16).padStart(2,"0")).join("");
      const groups=[];
      for(let i=0;i<16;i++){
        if(i%2===0){ const idx=bytes[i%bytes.length]%LEET_WORDS.length; groups.push(LEET_WORDS[idx]); }
        else { groups.push(hex.slice(i*4,(i+1)*4)); }
      }
      return groups.join(":");
    }
